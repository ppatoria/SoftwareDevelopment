#+begin_src c++
// File: const_1.h
const int ii = 100;// internal linkage

// File: const_2.h
const int ii = 100;// internal linkage

// File: using_const_1.cpp
#include <iostream>
#include "const_1.h"
using namespace std;
class cls{
public:
    int func(){ cout << ii << endl; }
};

// File: using_const_2.cpp
#include <iostream>
#include "const_2.h"
using namespace std;
class cls{
public:
    int func(){ cout << ii << endl; }
};
#+end_src
_Compile_

#+begin_src make
g++ -c using_const_1.cpp -o using_const_1.o  
g++ -c using_const_2.cpp -o using_const_2.o 
g++ using_const_1.o using_const_2.o -o using_const.exe
#+end_src

#+BEGIN_VERSE
*NOTE: const in c++  default to internal linkage.* 

_When (non-const) variables are defined in multiple headers._ 
*Example:* If ii is defines as non-const in const_1.h and const_1.h it would have lead to linker error as below. 
*Error:* multiple definition of ii
 #+END_VERSE
